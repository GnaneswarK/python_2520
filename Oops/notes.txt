#   How class looks like in Python

class Dog:
    def __init__(self, name):
        self.name = name

    def bark(self):
        print(f"{self.name} is barking!")

d = Dog("Tommy")
d.bark()

In Java we have 3 keyword public private protected
similary in python also we have these 3 keywords
by default if we dont mention anything then it is public
if we use one time underscore then it is protected 
if we use 2 times underscore then it is private
#   now we will see how to declare public private protected variables

def __init__(self,id,name,age):
    self.__id = id      #   private
    self.name = name    #   public
    self._age = age     #   protected
